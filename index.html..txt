    <script>
        const map = L.map('map').setView([37.8, -96], 4);
        L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);

        const data = {
            'US': [
                { name: "New York", coords: [40.7128, -74.0060], score: 84 },
                { name: "California", coords: [36.7783, -119.4179], score: 71 },
                { name: "Florida", coords: [27.6648, -81.5158], score: 52 }
            ],
            'Global': [
                { name: "France", coords: [46.2276, 2.2137], score: 76 },
                { name: "United Kingdom", coords: [55.3781, -3.4360], score: 45 }
            ]
        };

        function drawPins(type) {
            map.eachLayer(l => { if (l instanceof L.Marker) map.removeLayer(l); });
            data[type].forEach(loc => {
                const icon = L.divIcon({
                    className: 'danger-pin',
                    html: `<span class="pin-text">${loc.score}%</span>`,
                    iconSize: [35, 35]
                });
                L.marker(loc.coords, { icon: icon }).addTo(map).bindPopup(`${loc.name}: ${loc.score}% Risk`);
            });
            map.setView(type === 'US' ? [37.8, -96] : [20, 0], type === 'US' ? 4 : 2);
        }

        function toggleView(type) {
            drawPins(type);
            document.getElementById('btn-us').className = type === 'US' ? 'bg-red-600 px-4 py-2 rounded font-bold text-xs uppercase border border-red-400' : 'bg-gray-800 px-4 py-2 rounded font-bold text-xs uppercase border border-gray-600 text-gray-400';
            document.getElementById('btn-global').className = type === 'Global' ? 'bg-red-600 px-4 py-2 rounded font-bold text-xs uppercase border border-red-400' : 'bg-gray-800 px-4 py-2 rounded font-bold text-xs uppercase border border-gray-600 text-gray-400';
        }

        function sendMessage() {
            const input = document.getElementById('chat-input');
            if(!input.value) return;
            const chat = document.getElementById('chat-box');
            chat.innerHTML += `<div class="p-2 bg-gray-800 rounded mt-2"><span class="text-green-400 font-bold">User:</span> ${input.value}</div>`;
            input.value = '';
            chat.scrollTop = chat.scrollHeight;
        }

        drawPins('US');
    </script>
